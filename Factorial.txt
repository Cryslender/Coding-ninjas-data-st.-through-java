//This class will compute factorial

import java.util.*;                 //import all java libraries
public class Main
{
	public static void main(String[] args) 
	{
	    Scanner input = new Scanner(System.in);           //objects that will be used to get user input
		int n;                                          //declaring n to hold nuumber to compute its factorial
		System.out.println("Enter nonnegetive number to find its factorial: ");
		n= input.nextInt();
		System.out.println("factorial of" + n + "is: " + Fac(n));
		input.close();
	}
	
	//factorial method 
	public static int Fac(int n)
	{
	    int answer=0;
	    if(n==0)
	    {
	        answer = 1;
	    }
	    else
	    {
	        answer = n * Fac(n-1);
	    }
	    return answer;
	}
}
