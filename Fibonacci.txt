//This class will find the value in fibonacci series by its index

import java.util.*;                 //import all java libraries
public class Main
{
	public static void main(String[] args) 
	{
	    Scanner input = new Scanner(System.in);           //object that will be used to get user input
		int n;                                          //declaring n to hold index of the value in fibonacci series
		System.out.println("Enter index number(nonnegative) to find its value: ");
		n= input.nextInt();
		System.out.println("Value at index " + n + " is: " + Fib(n));
		input.close();
	}
	
	//fibonacci method to find the value with its index 
	public static int Fib(int index)
	{
	    int answer=0;
	    if(index==0)
	    {
	        answer =0;
	    }
	    else if(index==1)
	    {
	        answer = 1;
	    }
	    else
	    {
	        answer = Fib(index-1) + Fib(index-2);
	    }
	    return answer;
	}
}
